## IaC Tools / Công cụ IaC

### Example 1: Terraform / Ví dụ 1: Terraform

```hcl
# Terraform / Terraform
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 4.0"
    }
  }
}

resource "aws_instance" "app_server" {
  ami           = "ami-0c55b159cbfafe1f0"
  instance_type = "t2.micro"
  
  tags = {
    Name = "AppServer"
  }
}
```
